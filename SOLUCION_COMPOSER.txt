cd /var/www/html

# Mover Composer
sudo mv composer.phar /usr/local/bin/composer

# Verificar Composer
composer --version

# Instalar PHPMailer
composer require phpmailer/phpmailer

# Verificar PHPMailer
ls -la vendor/phpmailer/
